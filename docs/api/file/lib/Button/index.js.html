    /**
 * @file Button Component
 *
 * @author Leo Wang(wangkemiao@zhihu.com)
 */

import omit from 'lodash.omit'
import React, {Component} from 'react'
import PropTypes from 'prop-types'
import cx from 'classnames'

import {CLASS_PREFIX} from '../util/config'
import connectControl from '../Form/connectControl'

import './index.css'

/**
 * 按钮组件
 *
 * @export
 * @class Button
 * @extends {Component}
 */
@connectControl
export default class Button extends Component {

  /**
   * 组件支持传入的 props 定义，继承了 {@link Form.Control#propTypes} 指定的会传入真实控件的 props，故在此只介绍私有的 props
   *
   * @property {string} [type='button'] 类型，可选值同 html button 的 type 取值： button | submit | reset | menu（暂未实现）
   * @property {function(e:Event)} onClick 点击事件句柄
   */
  static propTypes = {
    type: PropTypes.string,
    onClick: PropTypes.func
  }

  /**
   * 默认的 props
   *
   * @static
   * @memberof Button
   */
  static defaultProps = {
    type: 'button',
    onClick(e) {
      e.preventDefault()
    }
  }

  /**
   * 点击行为处理
   *
   * @memberof Button
   *
   * @param {React.MouseEvent} e 事件参数
   */
  handleClick = e => {
    const {onClick} = this.props
    onClick(e)
  }

  render() {
    const {
      id,
      children,
      status,
      ...resetProps
    } = this.props

    const passThroughProps = omit(resetProps, ['onClick', 'onValueChange'])

    const classes = cx(
      `${CLASS_PREFIX}Button`,  // 控件样式
      `${CLASS_PREFIX}Button--is-${status}`  // 状态样式
    )
    return (
      <button
        id={id}
        name={id}  // name 与 id 相同
        {...passThroughProps}
        className={classes}
        onClick={this.handleClick}
      >
        {children}
      </button>
    )
  }
}
