    lib/Form/connectControl.js

    /**
 * @file FormControl 组件的 decorator
 *
 * @author Leo Wang(wangkemiao@zhihu.com)
 */

import isEqual from 'lodash.isequal'
import omit from 'lodash.omit'
import React, {Component} from 'react'
import PropTypes from 'prop-types'
import cx from 'classnames'
import kebabCase from 'lodash.kebabcase'
import AsyncValidator from 'async-validator';

import {getStatusFromProps} from '../util'
import {CLASS_PREFIX, STATUS} from '../util/config'

import './Control.css'

export default function connectControl(WrappedComponent) {

  /**
   * 表单元素组件
   *
   * @export FormControl
   */
  return class FormControl extends Component {

    /**
     * 组件接受并使用的 props 定义，所有组件的公共 props
     *
     * 注意在这里会传递给真实控件的 props 只有 id, status
     * 还有两个传过去的实际是 Control 接管的： value, onValueChange，并不是通过 props 传入的
     *
     * @static
     * @memberof Control
     *
     * @property {string} id 命名，此 id 会被传给真实控件，会被 Form 接管
     * @property {string} label 标签
     * @property {Array|Object} rules 校验规则，会被 Form 接管
     * @property {any} value 控件值，会被 Form 接管，直接修改无效
     * @property {any} defaultValue 控件默认值
     * @property {string} status 状态，会被 Form 接管
     * @property {string} message 提示信息，会被 Form 接管
     * @property {boolean} [disabled=false] 是否禁用
     * @property {boolean} [readonly=false] 是否只读
     * @property {string} className 从外部指定的自定义样式类名，不会传递给真实控件
     * @property {React.CSSProperties} style 从外部指定的自定义样式，不会传递给真实控件
     *
     * @property {function} onValueChange 值变化时的相应句柄，来自于 Form hijack
     * @property {function} onStatusChange status 变化时的相应句柄，来自于 Form hijack
     * @property {function} onRulesChange rules 变化时的相应句柄，来自于 Form hijack
     * @property {function} onMessageChange message 变化时的相应句柄，来自于 Form hijack
     */
    static propTypes = {
      id: PropTypes.string,
      label: PropTypes.string,
      className: PropTypes.string,
      style: PropTypes.object,

      // 会被 Form 自动灌入的事件句柄
      onValueChange: PropTypes.func,
      onStatusChange: PropTypes.func,
      onRulesChange: PropTypes.func,
      onMessageChange: PropTypes.func,
    }

    /**
     * 声明一个本类的 mark，当前用于 Form 中查找 Controls
     *
     * @static
     */
    static IS_ZHUI_FORM_CONTROL = true

    /**
     * Creates an instance of Control.
     * @param {Object} props
     * @memberof Control
     */
    constructor(props) {
      super(props)
      this.generateValidator(props)
    }

    /**
     * props 被修改时的处理
     *
     * @param {Object} nextProps
     *
     * @emits {onRulesChange} rule 变化时的响应句柄
     * @emits {onStatusChange} status 变化时的响应句柄
     */
    componentWillReceiveProps(nextProps) {
      const {id, onRulesChange, onStatusChange} = this.props
      // rule 被修改了，同步触发事件
      if (nextProps.rules !== this.props.rules
        || (nextProps.rules && this.props.rules && nextProps.rules.length !== this.props.rules.length)
        || !isEqual(nextProps.rules, this.props.rules)) {
        this.generateValidator(nextProps)
        onRulesChange && onRulesChange({
          payload: {
            [id]: nextProps.rules
          }
        })
      }
      // diabled, readonly 被修改，触发状态更改
      if (nextProps.disabled !== this.props.disabled || nextProps.readonly !== this.props.readonly) {
        const status = getStatusFromProps(nextProps, this.props.status)
        onStatusChange && onStatusChange({
          payload: {
            [id]: {status}
          }
        })
      }
    }

    /**
     * 根据 rules 生成校验使用的 validator
     *
     * @param {Object} [props=this.props]
     */
    generateValidator(props = this.props) {
      const {id, rules} = props
      this.validator = new AsyncValidator({
        [id]: rules || []
      })
    }

    /**
     * 控件值发生修改时的处理
     *
     * @method
     *
     * @param {any} value 修改后的值
     *
     * @emits {onValueChange} 值变化时的响应句柄
     */
    onValueChange = value => {
      const {id, onValueChange} = this.props

      onValueChange && onValueChange({
        payload: {
          [id]: value
        }
      })

      this.validate(value).catch(() => {})
    }

    /**
     * 执行校验
     *
     * @param {any} [value=this.props.value]
     *
     * @emits {onStatusChange} status 变化时的响应句柄
     * @emits {onMessageChange} message 变化时的响应句柄
     *
     * @returns {Promise}
     */
    validate(value = this.props.value) {
      const {id, onStatusChange, onMessageChange} = this.props
      return new Promise((resolve, reject) => {
        // validation
        if (this.validator) {
          this.validator.validate({[id]: value}, {first: true}, (errors, fields) => {
            if (errors) {
              reject(errors)
            }
            else {
              resolve()
            }
          })
        }
        else {
          resolve()
        }
      }).then(() => {
        onStatusChange && onStatusChange({
          payload: {
            [id]: STATUS.PASSED
          }
        })
        onMessageChange && onMessageChange({
          payload: {
            [id]: ''
          }
        })
      }).catch(errors => {
        const payload = {
          [id]: STATUS.FAULTY
        }
        onStatusChange && onStatusChange({
          payload: {
            [id]: STATUS.FAULTY
          }
        })
        onMessageChange && onMessageChange({
          payload: {
            [id]: errors[0].message
          }
        })
        throw payload
      })
    }

    /**
     * 渲染
     *
     * @memberof Control
     */
    render() {
      const {
        id,
        label,
        value,
        status = getStatusFromProps(this.props),
        message,
        className,
        style,
        children,
        ...resetProps
      } = this.props

      const passThroughProps = omit(resetProps, [
        'rules', 'defaultValue',
        'onValueChange', 'onStatusChange', 'onRulesChange', 'onMessageChange'
      ])

      const classes = cx(
        `${CLASS_PREFIX}FormCtrl`,  // 整体样式
        `${CLASS_PREFIX}FormCtrl--is-${status}`,  // 状态类
        className  // 外部样式
      )

      const labelClasses = cx(
        `${CLASS_PREFIX}FormCtrl-label`,  // 整体样式
        `${CLASS_PREFIX}FormCtrl-label--is-${status}`  // 状态类
      )

      const contentClasses = cx(
        `${CLASS_PREFIX}FormCtrl-content`,  // 整体样式
        `${CLASS_PREFIX}FormCtrl-content--is-${status}`  // 状态类
      )

      return (
        <div
          className={classes}
          style={style}
        >
          {label && (
            <label
              htmlFor={id}
              title={typeof label === 'string' ? label : ''}
              className={labelClasses}
            >
              {label}
            </label>
          )}
          <div className={contentClasses}>
            <WrappedComponent
              {...passThroughProps}
              id={id}
              value={value}
              status={status}
              onValueChange={this.onValueChange}
            >
              {children}
            </WrappedComponent>
            {message && <div className={`${CLASS_PREFIX}FormCtrl-message`}>{message}</div>}
          </div>
        </div>
      )
    }
  }
}
