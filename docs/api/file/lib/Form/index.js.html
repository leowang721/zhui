    /**
 * @file Form
 *
 * @author Leo Wang(wangkemiao@zhihu.com)
 */

import isEqual from 'lodash.isequal'
import AsyncValidator from 'async-validator';
import React, {Component} from 'react'
import PropTypes from 'prop-types'
import cx from 'classnames'

import {getStatusFromProps} from '../util'
import {CLASS_PREFIX, STATUS, LAYOUT} from '../util/config'
import FormGroup from './Group'

import './index.css'

/**
 * 表单组件
 *
 * @export
 * @class Form
 * @extends {Component}
 */
export default class Form extends Component {

  /**
   *
   * 组件接受并使用的 props 定义
   *
   * @static
   * @memberof Form
   *
   * @property {string} [layout=config.LAYOUT.HORIZONTAL] 布局
   * @property {string} className 从外部指定的自定义样式类名
   * @property {React.CSSProperties} style 从外部指定的自定义样式
   * @property {function(formValue: Object)} onSubmit 提交事件响应句柄
   */
  static propTypes = {
    layout: PropTypes.string,
    className: PropTypes.string,
    style: PropTypes.object,
    onSubmit: PropTypes.func
  }

  /**
   * 构造函数
   *
   * @param {any} props
   */
  constructor(props) {
    super(props)
    this.initializeState()
  }

  /**
   * 状态初始化，主要是根据 props.children 去初始化 values, rules, status, messages
   * 生成 state
   */
  initializeState() {
    const {children} = this.props
    const controls = this.getFormControls(children)

    const values = {}
    const rules = {}
    const status = {}
    const messages = {}

    for (let i = 0; i < controls.length; i++) {
      let ctrl = controls[i]
      values[ctrl.props.id] = ctrl.props.defaultValue
      if (ctrl.props.rules) {
        rules[ctrl.props.id] = ctrl.props.rules
      }
      status[ctrl.props.id] = getStatusFromProps(ctrl.props, ctrl.props.status)
      messages[ctrl.props.id] = ''
    }

    this.state = {values, rules, status, messages, initValues: Object.assign({}, values)}
  }

  /**
   * 是否修改了表单项的值
   *
   * @returns {boolean}
   */
  hasChanged() {
    return isEqual(this.state.values, this.state.initValues)
  }

  /**
   * 获取表单下所有的表单控件，被 Control 封过的
   *
   * @param {React.Children} children
   *
   * @returns {Array.<Object>} 子控件，注意是 Object
   */
  getFormControls(children) {
    let controls = []
    const childrenArray = React.Children.toArray(children)
    for (let i = 0; i < childrenArray.length; i++) {
      const child = childrenArray[i]

      if (!React.isValidElement(child)) {
        continue
      }
      if (child.type.IS_ZHUI_FORM_CONTROL) {
        controls.push(child)
      }
      else {
        if (child.props.children) {
          controls = controls.concat(this.getFormControls(child.props.children))
        }
      }
    }
    return controls
  }

  /**
   * 更新值
   *
   * @param {Object} toUpdate 要更新的信息，key 为 id，value 为值
   */
  updateControlValue(toUpdate) {
    this.setState({
      values: Object.assign(this.state.values, toUpdate)
    })
  }

  /**
   * 更新 status
   *
   * @param {Object} toUpdate 要更新的信息，key 为 id，value 为值
   */
  updateControlStatus(toUpdate) {
    this.setState({
      status: Object.assign(this.state.status, toUpdate)
    })
  }

  /**
   * 更新 rules
   *
   * @param {Object} toUpdate 要更新的信息，key 为 id，value 为值
   */
  updateControlRules(toUpdate) {
    this.setState({
      rules: Object.assign(this.state.rules, toUpdate)
    })
  }

  /**
   * 更新 message
   *
   * @param {Object} toUpdate 要更新的信息，key 为 id，value 为值
   */
  updateControlMessages(toUpdate) {
    this.setState({
      messages: Object.assign(this.state.messages, toUpdate)
    })
  }

 /**
  * 传递给子级控件的用于处理值变化的响应句柄
  *
  * @param {Object} e 参数
  * @param {Object} e.payload 变化信息，key 为 id，value 为值
  */
  onValueChange = e => {
    this.updateControlValue(e.payload)
  }

 /**
  * 传递给子级控件的用于处理 status 变化的响应句柄
  *
  * @param {Object} e 参数
  * @param {Object} e.payload 变化信息，key 为 id，value 为值
  */
  onStatusChange = e => {
    this.updateControlStatus(e.payload)
  }

 /**
  * 传递给子级控件的用于处理 rules 变化的响应句柄
  *
  * @param {Object} e 参数
  * @param {Object} e.payload 变化信息，key 为 id，value 为值
  */
  onRulesChange = e => {
    this.updateControlRules(e.payload)
  }

 /**
  * 传递给子级控件的用于处理 message 变化的响应句柄
  *
  * @param {Object} e 参数
  * @param {Object} e.payload 变化信息，key 为 id，value 为值
  */
  onMessageChange = e => {
    this.updateControlMessages(e.payload)
  }

  /**
   * 处理提交
   *
   * @param {Event} e 事件参数
   */
  handleSubmit = e => {
    this.validate().then(() => {
      const {onSubmit} = this.props
      onSubmit(this.state.values)
    }).catch(() => {})
    e.preventDefault()
  }

  /**
   * 整体校验
   *
   * @returns {Promise}
   */
  validate() {
    const validator = new AsyncValidator(this.state.rules)
    const payload = {
      status: {},
      messages: {}
    }
    return new Promise((resolve, reject) => {
      validator.validate(this.state.values, (errors, fields) => {
        if (errors) {
          reject(errors)
        }
        else {
          resolve()
        }
      })
    }).then(() => {
      for (var k in this.state.status) {
        payload.status[k] = STATUS.PASSED
        payload.messages[k] = ''
      }
      this.updateControlStatus(payload.status)
      this.updateControlMessages(payload.messages)
    }).catch(errors => {
      for (let i = 0; i < errors.length; i++) {
        let err = errors[i]
        payload.status[err.field] = STATUS.FAULTY;
        payload.messages[err.field] = err.message
      }

      this.updateControlStatus(payload.status)
      this.updateControlMessages(payload.messages)
      throw payload
    })
  }

  /**
   * 获取被劫持处理后的子级组件们，用于渲染
   *
   * @param {React.Children} children
   *
   * @returns {React.Children}
   */
  getHijackedChildren(children) {

    const {values, status, messages} = this.state
    const toExtendProps = {
      onValueChange: this.onValueChange,
      onStatusChange: this.onStatusChange,
      onRulesChange: this.onRulesChange,
      onMessageChange: this.onMessageChange
    }

    return React.Children.map(children, el => {
      if (React.isValidElement(el)) {
        if (el.type.IS_ZHUI_FORM_CONTROL) {
          const id = el.props.id
          return <el.type {...el.props} {...toExtendProps} value={values[id]} status={status[id]} message={messages[id]}>
            {this.getHijackedChildren(el.props.children)}
          </el.type>
        }
        return <el.type {...el.props}>
          {this.getHijackedChildren(el.props.children)}
        </el.type>
      }
      return el
    })
  }

  /**
   * 渲染
   */
  render() {
    const {
      layout = LAYOUT.HORIZONTAL,
      onSubmit,
      className,
      style,
      children
    } = this.props

    const classes = cx(
      `${CLASS_PREFIX}Form`,  // 整体样式
      `${CLASS_PREFIX}${LAYOUT[layout.toUpperCase()] || LAYOUT.HORIZONTAL}`,  // 布局控制
      className  // 外部样式
    )

    return (
      <form
        className={classes}
        style={style}
        onSubmit={this.handleSubmit}
      >
        {this.getHijackedChildren(children)}
      </form>
    )
  }
}

Form.Group = FormGroup

export const Group = FormGroup
