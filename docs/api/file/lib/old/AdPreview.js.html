    lib/old/AdPreview.js

    import React, {Component} from 'react'
import {connect} from 'react-redux'
import {bindActionCreators} from 'redux'
import AdPreviewComponent from 'react-zhihu-adpreview'
import * as imageActions from '../actions/image'
import './AdPreview.css'

/**
 * 这应该是个业务组件，并且这里还使用了 redux ~ actions 的模式
 *
 * @deprecated
 */
@connect(
  state => ({
    promotion: state.newAd.promotion,
  }),
  dispatch => ({
    actions: bindActionCreators(imageActions, dispatch)
  })
)
export default class AdPreview extends Component {

  /**
   * 点击上传按钮时的处理方法
   *
   * @param {string} templateName 模板名称
   * @param {blob[]} files 文件列表
   *
   * @emits {actions.imageUpload} 调用响应句柄 actions.imageUpload
   */
  onUploadImage = (templateName, files) => {
    this.props.actions.imageUpload(templateName, files[0], {type: 'ad'})
  }

  /**
   * 点击重置按钮时的处理方法
   *
   * @emits {actions.imageReset} 调用响应句柄 actions.imageReset
   */
  onResetImage = () => {
    this.props.actions.imageReset()
  }

  /**
   * 渲染
   *
   * @returns {ReactElement}
   */
  render() {
    const {template: {name}} = this.props
    const props = {
      ...this.props,
      onDrop: (files) => this.onUploadImage(name, files),
      onReset: this.onResetImage,
    }

    return AdPreviewComponent ? <AdPreviewComponent type={name} {...props} /> : null
  }
}
