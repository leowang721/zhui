    lib/old/Alert.js

    import React, {Component} from 'react'
import cx from 'classnames'
import {ALERT_DISMISS_TIMEOUT} from '../constants'
import './Alert.css'

/**
 * Alert 组件
 *
 * 实现有问题，使用了外部的 constant
 */
export default class Alert extends Component {

  /**
   * 默认的 props
   */
  static defaultProps = {
    onDissmiss() {}
  }

  /**
   * 计时器
   */
  timer = null

  /**
   * 设置消失计时，时间使用了外部的 constant.ALERT_DISMISS_TIMEOUT
   *
   * @emits {onDissmiss} 调用响应句柄 onDissmiss
   */
  setDismissTimer = () => {
    this.timer = setTimeout(() => {
      this.props.onDissmiss()
      this.timer = null
    }, ALERT_DISMISS_TIMEOUT)
  }

  /**
   * 清除消失即时
   */
  clearDismissTimer = () => {
    if (this.timer !== null) {
      clearTimeout(this.timer)
      this.timer = null
    }
  }

  componentDidMount() {
    this.setDismissTimer()
  }

  componentWillUnmount() {
    this.clearDismissTimer()
  }

  /**
   * 渲染
   *
   * @returns {ReactElement}
   */
  render() {
    const {msg, style} = this.props

    const className = cx('Alert', {
      'Alert--success': style === 'success',
      'Alert--warning': style === 'warning',
      'Alert--danger': style === 'danger'
    })

    return (
      <div className={className}>{msg}</div>
    )
  }
}
