    lib/old/Button.js

    import PropTypes from 'prop-types'
import React, {Component} from 'react'
import cx from 'classnames'
import './Button.css'

/**
 * 组件 Button
 */
export default class Button extends Component {

  /**
   * 组件支持传入的 props 定义
   *
   * @property {ReactElement} children 内嵌子组件，用这个展示内容
   * @property {string} className 可从外部指定样式类名
   * @property {boolean} [text=false] 是否是文本型，即长得像一段文本，实际上这个应该就是指 LinkButton 吧？
   * @property {boolean} [plain=false] 背景色为白色的普通按钮
   * @property {boolean} [danger=false] 是否是警示性按钮，边框和背景会变红，红色的按钮背景不应该是 danger 属性指定出来的
   * @property {boolean} [disabled=false] 是否禁用
   * @property {string} [type] 类型，可选值同 {@link button} 标签的 type 取值
   * @property {string} [target] 目标，可选值同 {@link a} 标签的 target 取值
   * @property {function(e:Event)} onClick 点击事件句柄
   */
  static propTypes = {
    children: PropTypes.node,
    text: PropTypes.bool,
    plain: PropTypes.bool,
    danger: PropTypes.bool,
    disabled: PropTypes.bool,
    type: PropTypes.string,
    onClick: PropTypes.func,
    target: PropTypes.string,
  }

  /**
   * 渲染
   *
   * @override
   *
   * @returns {ReactElement} React 元素
   */
  render() {
    const {
      text, plain, danger, children, className,
      component: Component='button', ...rest
    } = this.props

    return (
      <Component
        className={cx('Button-adv', className, {
          'Button-text': text,
          'Button-plain': plain,
          'Button-danger': danger,
        })}
        {...rest}
      >
        {children}
      </Component>
    )
  }
}
