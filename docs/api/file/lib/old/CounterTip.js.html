    lib/old/CounterTip.js

    import PropTypes from 'prop-types'
import React, {Component} from 'react'
import cx from 'classnames'
import './CounterTip.css'

/**
 * 一个展现 `计数` 的组件，形如 3/10 这种
 */
export default class CounterTip extends Component {

  /**
   * 组件支持传入的 props 定义
   *
   * @property {string} className 可从外部指定样式类名
   * @property {!number} length 当前计数
   * @property {!number} maxLength 总数
   */
  static propTypes = {
    length: PropTypes.number.isRequired,
    maxLength: PropTypes.number.isRequired,
    className: PropTypes.string,
  }

  /**
   * 默认的 props
   */
  static defaultProps = {
    length: 0,
    maxLength: 0
  }

  render() {
    const {
      length,
      maxLength,
      className,
    } = this.props

    if (maxLength > 0) {
      return (
        <span
          className={cx('CounterTip', className, {'CounterTip--error': length > maxLength})}
        >
          {`${length}/${maxLength}`}
        </span>
      )
    }
    return null
  }
}
