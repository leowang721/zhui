    lib/old/DataTable/index.js

    import PropTypes from 'prop-types'
import React from 'react'
import get from 'lodash/get'
import Tabs from '../Tabs'
import SearchBar from '../SearchBar'
import Spinner from '../Spinner'
import {NormalTh, FilterTh, SortTh} from './Th'
import './index.css'

const HeadingMap = {
  normal: NormalTh,
  filter: FilterTh,
  sort: SortTh,
}

function DataTableThead({renderers}) {
  return (
    <thead>
      <tr>
        {renderers.map((render) => {
          const Heading = HeadingMap[render.type || 'normal']
          return <Heading key={render.heading} {...render} />
        })}
      </tr>
    </thead>
  )
}

function DataTableTbody({renderers, data, rowProps}) {
  return (
    <tbody>
      {data.map((item, index) => (
        <tr
          key={index}
          {...rowProps(item)}
        >
          {renderers.map(renderer => (
            <td key={renderer.heading}>
              {renderer.dataField ? get(item, renderer.dataField) : renderer.render(item)}
            </td>
          ))}
        </tr>
      ))}
    </tbody>
  )
}

function Table({renderers, data, rowProps}) {
  return (
    <table className="DataTable-table">
      <DataTableThead renderers={renderers} />
      <DataTableTbody renderers={renderers} data={data} rowProps={rowProps} />
    </table>
  )
}

Table.propTypes = {
  data: PropTypes.arrayOf(PropTypes.object).isRequired,
  renderers: PropTypes.arrayOf(PropTypes.shape({
    heading: PropTypes.node.isRequired,
    thClassName: PropTypes.string,
    dataField: PropTypes.string,
    render: PropTypes.func,
    type: PropTypes.oneOf(['normal', 'filter', 'sort']),
    state: PropTypes.string,
    onStateChange: PropTypes.func,
  })).isRequired,
  rowProps: PropTypes.func,
}

export default function DataTable(props) {
  const {
    data, renderers, isLoading, rowProps,
    tabs, selectedTab, onTabChange,
    searchText, onSearchTextChange, searchPlaceholder,
  } = props

  return (
    <div className="DataTable">
      {Boolean(tabs) &&
        <div className="DataTable-actions">
          <Tabs
            className="DataTable-Tabs"
            tabs={tabs}
            selectedTab={selectedTab}
            onTabChange={onTabChange}
          />
          {Boolean(onSearchTextChange) &&
            <SearchBar
              className="DataTable-SearchBar"
              value={searchText}
              onChange={onSearchTextChange}
              placeholder={searchPlaceholder}
            />
          }
        </div>
      }
      <Table renderers={renderers} data={data} rowProps={rowProps} />
      {(!isLoading && (!data || !data.length)) && <div className="DataTable-placeholder">没有内容</div>}
      {isLoading && (
        <div className="DataTable-loading">
          <Spinner className="DataTable-spinner" />
        </div>
      )}
    </div>
  )
}

DataTable.defaultProps = {
  rowProps() {}
}

DataTable.propTypes = {
  isLoading: PropTypes.bool,
  data: PropTypes.arrayOf(PropTypes.object).isRequired,
  renderers: PropTypes.arrayOf(PropTypes.shape({
    heading: PropTypes.node.isRequired,
    thClassName: PropTypes.string,
    dataField: PropTypes.string,
    render: PropTypes.func,
    type: PropTypes.oneOf(['normal', 'filter', 'sort']),
    state: PropTypes.string,
    onStateChange: PropTypes.func,
  })).isRequired,
  rowProps: PropTypes.func,
  tabs: PropTypes.arrayOf(PropTypes.shape({
    value: PropTypes.string,
    label: PropTypes.string,
    meta: PropTypes.node,
  })),
  selectedTab: PropTypes.string,
  onTabChange: PropTypes.func,
  searchText: PropTypes.string,
  onSearchTextChange: PropTypes.func,
  placeholder: PropTypes.string,
}
