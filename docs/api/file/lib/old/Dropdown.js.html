    lib/old/Dropdown.js

    import PropTypes from 'prop-types'
import React, {Component, Children, cloneElement} from 'react'
import {findDOMNode} from 'react-dom'
import Button from './Button'
import Popover from './Popover'
import EventsJar from './EventsJar/index'
import SVGIcon from './SVGIcon'
import './Dropdown.css'

/**
 * 下拉（按钮）列表，更准确的说，类似于 Menu，与 {@link Select} 的区别：
 *  - 长得更像是菜单项
 *  - 不可能支持 复选
 *  - 目的是执行命令，不会更新展示
 */
class Dropdown extends Component {

  /**
   * 组件支持传入的 props 定义
   *
   * @property {ReactElement} children 内嵌子组件，用这个展示内容
   * @property {string} className 可从外部指定样式类名
   * @property {string} title 标题
   * @property {boolean} [disabled=false] 是否禁用
   * @property {boolean} plain 这个是给内置 Button 使用的，可参考 {@link Button.propTypes}
   */
  static propTypes = {
    children: PropTypes.node,
    className: PropTypes.string,
    title: PropTypes.string,
    disabled: PropTypes.bool,
    plain: PropTypes.bool,
  }

  /**
   * 内置状态对象
   */
  state = {
    showOptions: false
  }

  /**
   * 处理下拉项浮层的展现/隐藏切换
   */
  handleToggleDropdown = () => {
    this.setState({showOptions: !this.state.showOptions})
  }

  /**
   * 点击文档其他位置时隐藏下拉项浮层
   *
   * @param {Event} e 事件
   */
  handleDocumentClick = (e) => {
    if (!findDOMNode(this).contains(e.target)) {
      this.close()
    }
  }

  /**
   * 隐藏下拉项浮层
   */
  close = () => {
    this.setState({showOptions: false})
  }

  render() {
    const {children, title, disabled, plain, ...rest} = this.props
    const {showOptions} = this.state

    return (
      <div {...rest} className="Dropdown">
        <Button
          className="Dropdown-button"
          plain={plain}
          disabled={disabled}
          onClick={this.handleToggleDropdown}
        >
          {title}
          <SVGIcon
            name="arrow"
            size={24}
            fill="#0f88eb"
            style={{transform: `rotate(${showOptions ? -90 : 90}deg)`}}
          />
        </Button>
        <Popover isOpen={showOptions}>
          <EventsJar
            target={document}
            onClick={this.handleDocumentClick}
          />
          <div className="Dropdown-menu">
            {Children.map(Children.toArray(children), (child) =>
              cloneElement(child, {
                onClick: () => {
                  if (child.props.onClick) { child.props.onClick() }
                  this.close()
                }
              })
            )}
          </div>
        </Popover>
      </div>
    )
  }
}

export default Dropdown
