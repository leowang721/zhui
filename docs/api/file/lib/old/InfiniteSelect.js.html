    lib/old/InfiniteSelect.js

    import PropTypes from 'prop-types'
import React, {Component} from 'react'
import cx from 'classnames'
import SVGIcon from './SVGIcon'
import Popover from './Popover'
import InfiniteScroll from './InfiniteScroll'
import EventsJar from './EventsJar'
import MenuItem from './MenuItem'
import './InfiniteSelect.css'

/**
 * 下拉选项，在滚动至底部时可触发加载之类的处理
 */
class InfiniteSelect extends Component {

  /**
   * 组件支持传入的 props 定义
   *
   * @property {string} className 可从外部指定样式类名
   * @property {array} options 下拉选择项的数据配置
   * @property {any} value 当前选择的值
   * @property {function(e:Event)} 选择触发值改变时的响应句柄
   * @property {function} 滚动到底部时的响应句柄
   */
  static propTypes = {
    className: PropTypes.string,
    options: PropTypes.array,
    value: PropTypes.any,
    onChange: PropTypes.func,
    onEndReached: PropTypes.func,
  }

  /**
   * 初始 state
   */
  state = {
    expand: false,
  }

  /**
   * 切换下拉项的展现
   */
  handleToggleExpand = () => {
    this.setState({expand: !this.state.expand})
  }

  /**
   * 处理选择某一下拉项
   *
   * @param {object} item 当前被选择的项
   * @param {any} item.value 取值
   *
   * @emits {onChange} 调用响应句柄 onChange
   */
  handleSelect = (item) => {
    if (this.props.value !== item.value) {
      this.props.onChange(item.value)
    }
    this.close()
  }

  /**
   * 关闭下拉项
   */
  close = () => {
    this.setState({expand: false})
  }

  render() {
    const {
      className, options, value, onEndReached, hasMore, title, isLoading
    } = this.props
    const {expand} = this.state
    const selectedItem = options.find((option) => option.value === value)
    const isEmpty = options.length === 0

    return (
      <div
        className={cx('InfiniteSelect', className, {'is-disabled': isEmpty})}
        onClick={this.handleToggleExpand}
      >
        {selectedItem ? selectedItem.title : title}
        <SVGIcon name="select" width={8} fill="#c4cdd9" />
        <Popover isOpen={expand}>
          <EventsJar target={document} onClick={this.close} />
          <InfiniteScroll
            className="InfiniteSelect-options"
            distance={20}
            onReachBottom={onEndReached}
          >
            {options.map((item, idx) => (
              <MenuItem
                key={idx}
                dark={item.value === value}
                onClick={() => this.handleSelect(item)}
              >
                {item.title}
              </MenuItem>
            ))}
            {isLoading && <MenuItem className="InfiniteSelect-tip">正在加载...</MenuItem>}
            {!isEmpty && !hasMore && <MenuItem className="InfiniteSelect-tip">没有更多了</MenuItem>}
          </InfiniteScroll>
        </Popover>
      </div>
    )
  }
}

export default InfiniteSelect
