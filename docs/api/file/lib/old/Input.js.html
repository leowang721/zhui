    lib/old/Input.js

    import PropTypes from 'prop-types'
import React, {Component} from 'react'
import cx from 'classnames'
import './Input.css'

export default class Input extends Component {

  /**
   * 组件支持传入的 props 定义
   *
   * @property {object} style 样式定义
   * @property {boolean} [spread=false] 是否宽度延展为 100%
   * @property {boolean} [right=false] 是否右对齐
   * @property {string} addon 输入框的单位
   * @property {ReactElement} error 错误信息 Element，error 和 helper 只能展现其一 error 优先级更高
   * @property {ReactElement} helper 帮助信息 Element，优先级低于 error
   */
  static propTypes = {
    style: PropTypes.object,
    spread: PropTypes.bool,
    right: PropTypes.bool,
    addon: PropTypes.string,
    error: PropTypes.node,
    helper: PropTypes.node,
  }

  /**
   * 实际的输入表单
   */
  input = null

  /**
   * 设置焦点
   */
  focus = () => {
    this.input.focus()
  }

  render() {
    const {
      className,
      spread,
      right,
      addon,
      error,
      helper,
      ...rest
    } = this.props

    return (
      <div className={cx('Input-adv', className, {
        'Input--right': right,
        'Input--spread': spread,
        'Input--addon': addon,
        'Input--error': error,
      })}>
        <input
          className="Input-field"
          ref={(node) => { this.input = node }}
          {...rest}
        />
        {addon && <span className="Input-addon">{addon}</span>}
        {(error || helper) &&
          <span className={cx('Input-message', {'Input-message--error': error})}>
            {error || helper}
          </span>
        }
      </div>
    )
  }
}
