    lib/old/Popover.js

    import PropTypes from 'prop-types'
import React, {Component} from 'react'
import cx from 'classnames'
import './Popover.css'

/**
 * 弹出浮层，类似于 Bubble
 */
export default class Popover extends Component {

  /**
   * 组件支持传入的 props 定义
   *
   * @property {ReactElement} children 内嵌子组件，用这个展示内容
   * @property {boolean} arrow 是否有小箭头
   * @property {!boolean} isOpen 是否开启状态
   */
  static propTypes = {
    arrow: PropTypes.bool,
    children: PropTypes.node,
    isOpen: PropTypes.bool.isRequired,
  }

  /**
   * 默认的 props
   */
  static defaultProps = {
    arrow: false,
  }

  /**
   * 点击时的处理句柄
   *
   * @param {Event} e 点击事件
   */
  handleClick = (e) => {
    if (e.target.tagName !== 'A') {
      e.stopPropagation()
      e.nativeEvent.stopImmediatePropagation()
    }
  }

  render() {
    const {children, arrow, isOpen, className} = this.props

    return isOpen ? (
      <div className={cx('Popover-adv', className)} onClick={this.handleClick}>
        {arrow && <div className="Popver-arrow" />}
        {children}
      </div>
    ) : null
  }
}
