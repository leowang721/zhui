    lib/old/Tooltip.js

    import PropTypes from 'prop-types'
import React, {Component, cloneElement} from 'react'
import cx from 'classnames'
import {
  findDOMNode,
  unmountComponentAtNode,
  unstable_renderSubtreeIntoContainer as renderSubtreeIntoContainer
} from 'react-dom'
import {isMouseEnterOrLeave} from '../lib/dom'
import Portal from './Portal'
import './Tooltip.css'

/**
 * 以浮层形式弹出的帮助组件
 */
class Tooltip extends Component {

  /**
   * 组件支持传入的 props 定义
   *
   * @property {string} 【position=“right”】 位置
   * @property {boolean} [clickable=false] 是否是点击触发
   * @property {boolean} [hoverable=true] 是否是悬停触发
   * @property {ReactElement} tooltip 具体信息的 Element
   */
  static propTypes = {
    position: PropTypes.string,
    clickable: PropTypes.bool,
    hoverable: PropTypes.bool,
    tooltip: PropTypes.node,
  }

   /**
    * 默认的 props
    */
  static defaultProps = {
    position: 'right',
    clickable: false,
    hoverable: true,
    tooltip: null,
  }

   /**
    * 初始 state
    */
  state = {
    show: false
  }

  /**
   * @type {?ReactElement} 具体的展现 Element 是一个 Portal
   */
  overlay = null

  /**
   * @type {?ReactElement} 展现内容所挂载的容易
   */
  mountDom = null

  /**
   * @type {object} 这个其实是设置蒙版大小的
   */
  targetOffset = {}

  componentDidMount() {
    this.mountDom = document.createElement('div')
    this.renderOverlay()
    this.targetOffset = this.getOffset()
  }

  componentDidUpdate() {
    this.renderOverlay()
    this.targetOffset = this.getOffset()
    if (this.props.clickable) {
      document.addEventListener('click', this.handleDocumentClick)
    }
  }

  componentWillUnmount() {
    unmountComponentAtNode(this.mountDom)
    this.mountDom = null
    if (this.props.clickable) {
      document.removeEventListener('click', this.handleDocumentClick)
    }
  }

  /**
   * 点击文档其他位置的隐藏处理
   */
  handleDocumentClick = (e) => {
    const trigger = findDOMNode(this)
    if (this.tooltip && !trigger.contains(e.target)) {
      this.setState({show: false})
    }
  }

  /**
   * 文档滚动时隐藏处理
   */
  handleDocumentScroll = () => {
    this.setState({show: false})
  }

  /**
   * 透传处理函数给子级元素处理
   */
  excuteChildEventHandler = (handler, event) => {
    const {children} = this.props
    if (children.props[handler]) {
      children.props[handler](event)
    }
  }

  /**
   * 鼠标悬停时，展现，并透传给子级元素处理
   */
  handleMouseOver = (event) => {
    this.targetOffset = this.getOffset()
    if (this.props.hoverable && isMouseEnterOrLeave(event)) {
      this.setState({show: true})
    }
    this.excuteChildEventHandler('onMouseOver', event)
  }

  /**
   * 鼠标移出时，隐藏，并透传给子级元素处理
   */
  handleMouseOut = (event) => {
    if (this.props.hoverable && isMouseEnterOrLeave(event)) {
      this.setState({show: false})
    }
    this.excuteChildEventHandler('onMouseOut', event)
  }

  /**
   * 鼠标点击时，切换展现，并透传给子级元素处理
   */
  handleClick = () => {
    this.targetOffset = this.getOffset()
    if (this.props.clickable) {
      this.setState({show: !this.state.show})
    }
    this.excuteChildEventHandler('onClick', event)
  }

  /**
   * 获取当前容器的大小
   */
  getOffset = () => {
    return findDOMNode(this).getBoundingClientRect()
  }

  /**
   * 产出展示内容元素
   */
  makeOverlay = () => {
    const {position, tooltip, className, smooth} = this.props
    const style = this.getPositionStyle(position)

    return this.state.show ? (
      <Portal>
        <div className={cx(
          'Tooltip-adv',
          `Tooltip--${position}`,
          className,
          {'Tooltip-smooth': smooth},
        )} style={style}>
          <div className="Tooltip-content" ref={(node) => { this.tooltip = node }}>
            {tooltip}
          </div>
        </div>
      </Portal>
    ) : <noscript />
  }

  /**
   * 渲染要展示的内容
   */
  renderOverlay = () => {
    renderSubtreeIntoContainer(this, this.overlay, this.mountDom)
  }

  /**
   * 计算位置相关的样式
   */
  getPositionStyle = (position) => {
    const style = {}
    const targetOffset = this.targetOffset

    const offsetTop = targetOffset.top
    const offsetBottom = targetOffset.bottom

    switch (position) {
      case 'top': {
        style.top = `${offsetTop}px`
        style.left = `${targetOffset.left + targetOffset.width / 2}px`
        break
      }
      case 'bottom': {
        style.top = `${offsetBottom}px`
        style.left = `${targetOffset.left + targetOffset.width / 2}px`
        break
      }
      case 'left': {
        style.left = `${targetOffset.left}px`
        style.top = `${offsetTop + targetOffset.height / 2}px`
        break
      }
      case 'right': {
        style.left = `${targetOffset.right}px`
        style.top = `${offsetTop + targetOffset.height / 2}px`
        break
      }
      default:
        break
    }
    return style
  }

  render() {
    const {children} = this.props
    const triggerProps = {
      onMouseOver: this.handleMouseOver,
      onMouseOut: this.handleMouseOut,
      onClick: this.handleClick,
    }

    this.overlay = this.makeOverlay()
    return cloneElement(children, triggerProps)
  }
}

export default Tooltip
