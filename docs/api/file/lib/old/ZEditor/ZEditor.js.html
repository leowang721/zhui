    import React, {Component} from 'react'
import {EditorState, RichUtils} from 'draft-js'
import SVGIcon from '../SVGIcon'
import RichInput from 'zhihu-components/es/RichInput'
import {ToolbarButton} from 'zhihu-components/es/Editable/Toolbar'
import Modal, {ModalButtonGroup} from 'zhihu-components/es/Modal'
import Button from 'zhihu-components/es/Button'
import Input from 'zhihu-components/es/Input'
import './ZEditor.css'

class PromotionEditor extends Component {
  state = {
    url: '',
    shown: false,
  }

  getToolbarPlugins = (editorComponent, {currentStyle}) => {
    this.editorComponent = editorComponent

    return [
      <ToolbarButton
        key="underline"
        icon="underline"
        label="下划线"
        isActive={currentStyle.has('UNDERLINE')}
        onClick={() => {
          editorComponent.onChange(
            RichUtils.toggleInlineStyle(
              editorComponent.state.editorState,
              'UNDERLINE'
            ),
            () => { editorComponent.editor.focus() }
          )
        }}
      />,
      <ToolbarButton
        key="link"
        icon={
          <SVGIcon
            name="link" height={18} width={18} fill="#9fadc7"
            style={{top: '4px', position: 'relative'}}
          />
        }
        label="添加链接"
        onClick={() => {
          this.setState({shown: true})
        }}
      />,
    ]
  }

  handleUrlChange = (e) => {
    this.setState({url: e.target.value})
  }

  handleUrlModalClose = () => {
    this.setState({shown: false})
  }

  setLink = (e) => {
    e.preventDefault()
    const {editorState} = this.editorComponent.state
    const contentState = editorState.getCurrentContent()
    const contentStateWithEntity = contentState.createEntity(
      'link',
      'MUTABLE',
      {url: this.state.url}
    )

    const entityKey = contentStateWithEntity.getLastCreatedEntityKey()
    const newEditorState = EditorState.push(editorState, contentStateWithEntity, 'apply-entity')
    this.editorComponent.onChange(
      RichUtils.toggleLink(
        newEditorState,
        newEditorState.getSelection(),
        entityKey
      ), () => { this.editorComponent.editor.focus() })

    this.setState({shown: false, url: ''})
  }

  render() {
    const {editorState, url, shown} = this.state

    return (
      <div>
        <RichInput
          {...this.props}
          allowVideoUpload
          spread
          large
          minRows={10}
          multiline
          allowToolbar
          allowToolbarSticky
          editorState={editorState}
          getToolbarPlugins={this.getToolbarPlugins}
          videoWatermark={false}
        />
        <Modal onClose={this.handleUrlModalClose}>
          {shown &&
          <form
            style={{textAlign: 'center', paddingTop: 30}}
            onSubmit={this.setLink}
          >
            <Input
              required
              spread
              type="url"
              placeholder="请输入链接地址"
              value={url}
              onChange={this.handleUrlChange}
            />

            <ModalButtonGroup>
              <Button
                preset="primary"
                color="blue"
                type="submit"
              >
                确认
              </Button>
              <Button
                onClick={this.handleUrlModalClose}
              >
                取消
              </Button>
            </ModalButtonGroup>
          </form>
          }
        </Modal>
      </div>
    )
  }
}

export default ({onChange, onImageUpload, onVideoUpload, ...props}) => {
  const newOnChange = (editorState) => {
    const raw = editorState.isEmpty ? null : editorState.raw
    onChange(raw)
  }

  const newOnImageUpload = (file) => {
    // Template Name differ
    return onImageUpload('editor', [file])
  }

  return (
    <PromotionEditor
      onChange={newOnChange}
      onImageUpload={newOnImageUpload}
      onVideoUpload={onVideoUpload}
      {...props} />
  )
}
